# fetch ImageData with datatime format
SELECT fileName, 
	grayImage, 
	rgbImage, 
	uniqueId, 
	to_char(datetime, 'DD Mon YYYY, HH:MI AM') 
FROM ImageData
WHERE userid = 1;



# query to fetch data in batches
SELECT 
	fileName, 
	grayImage, 
	rgbImage, 
	uniqueId, 
	to_char(datetime, 'DD Mon YYYY, HH:MI AM') 
FROM ImageData
WHERE userid = 1
LIMIT 4
OFFSET 0    // page no
